11   def get_model(): 
12       Input_1 = Input(shape=(1, 28, 28)) 
13       Conv2D_6 = Conv2D(nb_row= 3,border_mode= 'same' ,activation= 'relu' ,nb_col= 3,nb_filter= 32)(Input_1) 
14       MaxPooling2D_4 = MaxPooling2D(pool_size= (2,2),border_mode= 'same' ,strides= (2,2))(Conv2D_6) 
15       Dropout_1 = Dropout(p= 0.3)(MaxPooling2D_4) 
16       Conv2D_8 = Conv2D(nb_row= 3,border_mode= 'same' ,activation= 'relu' ,nb_col= 3,nb_filter= 64,init= 'glorot_normal' )(Dropout_1) 
17       MaxPooling2D_6 = MaxPooling2D(pool_size= (2,2),border_mode= 'same' ,strides= (2,2))(Conv2D_8) 
18       Dropout_2 = Dropout(p= 0.3)(MaxPooling2D_6) 
19       Conv2D_9 = Conv2D(nb_row= 3,border_mode= 'same' ,activation= 'relu' ,nb_col= 3,nb_filter= 128,init= 'glorot_normal' )(Dropout_2) 
20       MaxPooling2D_7 = MaxPooling2D(pool_size= (2,2),border_mode= 'same' )(Conv2D_9) 
21       Flatten_2 = Flatten()(MaxPooling2D_7) 
22       Dropout_3 = Dropout(p= 0.3)(Flatten_2) 
23       Dense_4 = Dense(activation= 'relu' ,init= 'glorot_normal' ,output_dim= 625)(Dropout_3) 
24       Dropout_4 = Dropout(p= 0.5)(Dense_4) 
25       Dense_5 = Dense(activation= 'softmax' ,output_dim= 10)(Dropout_4)
